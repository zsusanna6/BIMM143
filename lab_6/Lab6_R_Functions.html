We can also modify the grade() function to make drop lowest optional

grade2 <- function(x, drop.low = T){
  

  ##find NAs in vector and make them 0
  x[is.na(x)] <- 0
  
  if(drop.low){
    
    ##drop the lowest and finds the mean
    out <- mean(x[-which.min(x)])
  }else{
    out <- mean(x)
  }
  return(out)
}
